def dirReduc(arr):
    for i in range(len(arr)):
        if len(arr)-1 <= i or len(arr) == 1:
            return arr
        if len(arr) > 1 and ((arr[i] == "NORTH" and arr[i+1] == "SOUTH") \
                            or (arr[i] == "SOUTH" and arr[i+1] == "NORTH") \
                            or (arr[i] == "WEST" and arr[i+1] == "EAST") \
                            or (arr[i] == "EAST" and arr[i+1] == "WEST")):
            del arr[i+1]
            del arr[i]
            dirReduc(arr)
    return arr
